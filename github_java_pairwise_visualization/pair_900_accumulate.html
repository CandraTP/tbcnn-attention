public static int compare(String a, String b){
        if(a.length() < b.length())
            return -1;
        else if(a.length() > b.length())
            return 1;
        else
            return a.compareTo(b);
    }

    
    public static long sort(String[] A){
        
        long startTime = System.nanoTime();
        
        
        quick_sort(A, 0, A.length - 1);

        
        long endTime = System.nanoTime();
        return endTime - startTime;
    }

    
    public static int partition(String[] A, int p, int r){
        String x = A[r];
        String tmp;
        int i = p - 1;
        for(int j = p; j < r; j++){
            if(compare(A[j], x) <= 0){
                i++;
                
                
                tmp = A[j];
                A[j] = A[i];
                A[i] = tmp;
            }
        }
        i++;
        A[r] = A[i];
        A[i] = x;
        return i;
    }

    public static void quick_sort(String[] A, int p, int r){
        if(p < r){
            int q = partition(A, p, r);
            quick_sort(A, p, q-1);
            quick_sort(A, q+1, r);
        }
    }




public static int getmin(ArrayList<Integer> a){
	int retVal = a.get(0);
	a.set(0, a.get(a.size()-1));
	a.remove(a.size()-1);
	if(!a.isEmpty()) siftd(a,0);
	
	return retVal;
}

public static boolean verify(ArrayList<Integer> a){
	
	for(int i = 0; i < a.size()-1; i++){
		if(a.get(i) > a.get(i+1)) return false;
	}
	return true;
	
}


public static void heapify(ArrayList<Integer> a){
	for(int i = a.size()/2; i >=0 ; i--){
		siftd(a,i);
	}
	
}

public static void siftd(ArrayList<Integer> a, int i){
	int left = i*2;
	int right = 2*i +1;
	int smallest = i;
	
	if(left < a.size() && a.get(left) < a.get(i) ){
		smallest = left;
	}
	if(right < a.size() && a.get(right) < a.get(smallest) ){
		smallest = right;
	}
	if(smallest != i){
		int temp = a.get(i);
		a.set(i,a.get(smallest));
		a.set(smallest, temp);
		siftd(a,smallest);
	}
}
