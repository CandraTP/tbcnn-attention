public static void sort(Comparable[] a) {
        
        final int N  = a.length; int h = 1;
        while(h < N/3) h = 3*h + 1;
        
        
        while(h >= 1) {
        
            
            
            for(int i = h; i < N; i++) {
            
                
                
                for(int j = i; j >= h && less(a[j], a[j-h]); j -= h) exch(a, j, j-h);
                
            }
            h /= 3;
        }
        
        
    }


  
    int sort(int arr[])
    {
        int n = arr.length;
 
        
        for (int gap = n/2; gap > 0; gap /= 2)
        {
            
            
            
            
            for (int i = gap; i < n; i += 1)
            {
                
                
                
                int temp = arr[i];
 
                
                
                int j;
                for (j = i; j >= gap && arr[j - gap] > temp; j -= gap)
                    arr[j] = arr[j - gap];
 
                
                
                arr[j] = temp;
            }
        }
        return 0;
    }
