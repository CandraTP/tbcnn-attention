public static void mergeSort(int[] arr) {
      if (arr.length > 1) {
         int[] left = leftHalf(arr);
         int[] right = rightHalf(arr);

         mergeSort(left);
         mergeSort(right);

         mergeArrays(arr, left, right);
      }
   }

   public static int[] leftHalf(int[] arr) {
      int size = arr.length / 2;
      int[] left = new int[size];
      for (int i = 0; i < size; i++) {
         left[i] = arr[i];
      }
      return left;
   }

   public static int[] rightHalf(int[] arr) {
      int size1 = arr.length / 2;
      int size2 = arr.length - size1;
      int[] right = new int[size2];
      for (int i = 0; i < size2; i++) {
         right[i] = arr[i + size1];
      }
      return right;
   }

   public static void mergeArrays(int[] arr, int[] left, int[] right) {
      int i1 = 0;
      int i2 = 0;

      for (int i = 0; i < arr.length; i++) {
         if (i2 >= right.length || (i1 < left.length && left[i1] <= right[i2])) {
            arr[i] = left[i1];
            i1++;
         } else {
            arr[i] = right[i2];
            i2++;
         }
      }
   }


package PukyungUniv;

import static org.junit.Assert.assertEquals;

import java.util.Arrays;

public class Sort_HeapSort {

    private static int getLeftChildIdx(int parent) {
        return (parent + 1) * 2 - 1;
    }

    private int getParentIdx(int child) {
        if (child < 1) {
            return -1;
        } else {
            return (child + 1) / 2 - 1;
        }
    }

    private static void maxHeapify(int[] data, int i, int end) {
        int leftChild = getLeftChildIdx(i);
        while (leftChild <= end) {
            int largeChild;
            if (leftChild == end) {
                largeChild = leftChild;
            } else {
                largeChild = data[leftChild] > data[leftChild + 1] ? leftChild : leftChild + 1;
            }
            if (data[i] >= data[largeChild]) {
                return;
            }
            swap(data, i, largeChild);
            i = largeChild;
            leftChild = getLeftChildIdx(i);
        }
    }

    private static void swap(int[] data, int i, int j) {
        int t = data[i];
        data[i] = data[j];
        data[j] = t;
    }

    public static void sort(int[] data) {
        makeHeap(data);
        sortFromHeap(data);
    }

    private static void makeHeap(int[] data) {
        for (int i = data.length / 2 - 1; i >= 0; i--) {
            maxHeapify(data, i, data.length - 1);
        }
    }

    private static void sortFromHeap(int[] data) {
        int last = data.length - 1;
        while (last > 0) {
            swap(data, 0, last);
            last--;
            maxHeapify(data, 0, last);
        }
    }

}
