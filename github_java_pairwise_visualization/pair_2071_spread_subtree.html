void ordenaArray(int iz, int de) {

        int i, j, x, w;

        i = iz;
        j = de;

        x = vector[(iz + de) / 2];

        do {

            while (vector[i] < x) {
                i++; 
            }

            while (x < vector[j]) {
                j--; 
            }

            if (i <= j) {

                w = vector[i];

                vector[i] = vector[j];

                vector[j] = w;
                i++;  
                j--; 
            }
        } while (i <= j);


        w = vector[i];
        vector[i] = vector[de];
        vector[de] = w;

        if (iz < j) {
            ordenaArray(iz, j);
        }
        if (i < de) {
            ordenaArray(i, de);
        }
    }


 public static void sort(Comparable[] a) {
    int n = a.length;

    
    int h = 1;
    while (h < n/3) h = 3*h + 1;

    while (h >= 1) {
      
      for (int i = h; i < n; i++) {
        for (int j = i; j >= h && less(a[j], a[j-h]); j -= h) {
          exch(a, j, j-h);
        }
      }
      h /= 3;
    }
  }

  private static boolean less(Comparable v, Comparable w) {
    return v.compareTo(w) < 0;
  }

  private static void exch(Object[] a, int i, int j) {
    Object swap = a[i];
    a[i] = a[j];
    a[j] = swap;
  }
