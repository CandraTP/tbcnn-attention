public static int[] swap(int numbers[], int i, int j) {
        int temp = numbers[i];
        numbers[i] = numbers[j];
        numbers[j] = temp;
        return numbers;
    }

    public static int[] sort(int array[]) {
        for (int index = 1; index < array.length; index++) {
            for (int indexj = 0; indexj < index-1; indexj++)
                if (array[indexj] > array[index]) {
                    array = swap(array, indexj, index);
                }
        }
        return array;
    }



 public void sort(int a[]) {

        if(a == null || a.length <= 1) {
            return;
        }

        insertNumberAtIndexInSortedArray(a, 1);
    }

    private void insertNumberAtIndexInSortedArray(int a[], int index) {

        if (index == a.length) {
            return;
        }

        int number = a[index];
        int i;
        for (i=index-1; i >= 0; i--) {
            if (number < a[i]) {
                a[i+1] = a[i];
            } else {
                break;
            }
        }

        a[i+1] = number;

        insertNumberAtIndexInSortedArray(a, index+1);
    }
