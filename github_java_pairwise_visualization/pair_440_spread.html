public static int[] swap(int numbers[], int i, int j) {
        int temp = numbers[i];
        numbers[i] = numbers[j];
        numbers[j] = temp;
        return numbers;
    }

    public static int[] sort(int array[]) {
        for (int index = 1; index < array.length; index++) {
            for (int indexj = 0; indexj < index-1; indexj++)
                if (array[indexj] > array[index]) {
                    array = swap(array, indexj, index);
                }
        }
        return array;
    }

        int i, j, x, w;

        do {

            while (vector[i] < x) {
                i++; 
            }

            while (x < vector[j]) {

            }

            if (i <= j) {

                w = vector[i];

                vector[i] = vector[j];

                vector[j] = w;
                i++;  
                j--; 
            }
        } while (i <= j);


        w = vector[i];
        vector[i] = vector[de];
        vector[de] = w;

        if (iz < j) {
            ordenaArray(iz, j);
        }
        if (i < de) {
            ordenaArray(i, de);
        }
