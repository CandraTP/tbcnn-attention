void bubbleSort(int arr[]) 
    { 
        int n = arr.length; 
        for (int i = 0; i < n-1; i++) 
            for (int j = 0; j < n-i-1; j++) 
                if (arr[j] > arr[j+1]) 
                { 
                    
                    int temp = arr[j]; 
                    arr[j] = arr[j+1]; 
                    arr[j+1] = temp; 
                } 
    } 


public static void selsort(int a[]){
        int n=a.length;
        int temp;
        int pos=0;
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                if(a[i]>a[j])
                    pos=j;
            }
           temp=a[i];
           a[i]=a[pos];
           a[pos]=temp;
        }
      
         System.out.println("the sorted array is ");
         for(int i=0;i<n;i++){
             System.out.print(a[i]+ " ");
         }
    }
