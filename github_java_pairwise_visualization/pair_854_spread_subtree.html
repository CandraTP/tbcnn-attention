public static int[] swap(int numbers[], int i, int j) {
        int temp = numbers[i];
        numbers[i] = numbers[j];
        numbers[j] = temp;
        return numbers;
    }

    public static int[] sort(int array[]) {
        for (int index = 1; index < array.length; index++) {
            for (int indexj = 0; indexj < index-1; indexj++)
                if (array[indexj] > array[index]) {
                    array = swap(array, indexj, index);
                }
        }
        return array;
    }




    public static void sort(Comparable[] a) {
        
        final int N  = a.length; int h = 1;
        while(h < N/3) h = 3*h + 1;
        
        
        while(h >= 1) {
        
            
            
            for(int i = h; i < N; i++) {
            
                
                
                for(int j = i; j >= h && less(a[j], a[j-h]); j -= h) exch(a, j, j-h);
                
            }
            h /= 3;
        }
        
        
    }
