private static void selectionsort(double tablica[], int ile_liczb) {
    int min,i,j;
    double temp;
    for (i=0;i<ile_liczb-1;i++)  {
    min=i;
    for (j=i+1;j<ile_liczb;j++)
    if (tablica[j]<tablica[min])
    min=j;
    temp=tablica[min];
    tablica[min]=tablica[i];
    tablica[i]=temp;
    }
}


public static void sort(int[] data) {
        for (int k = data.length - 1; k > 0; k--) {
            int largestIdx = 0;
            for (int i = 0; i <= k; i++) {
                if (data[i] > data[largestIdx]) {
                    largestIdx = i;
                }
            }
            int tmp = data[k];
            data[k] = data[largestIdx];
            data[largestIdx] = tmp;
        }
    }

}
