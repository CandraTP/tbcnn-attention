public static void bubbleSort(int arr[]) {
      int len = arr.length;
      for (int i = len; i >= 0; i--) {
         for (int j = 0; j < len - 1; j++) {
            int k = j + 1;
            if (arr[j] > arr[k]) {
               swapNumbers(j, k, arr);
            }
         }
      }
   }

   private static void swapNumbers(int i, int j, int[] arr) {
      int temp;
      temp = arr[i];
      arr[i] = arr[j];
      arr[j] = temp;
   }



    public static void buildheap(int []a){
        n=a.length-1;
        for(int i=n/2;i>=0;i--){
            maxheap(a,i);
        }
    }
    
    public static void maxheap(int[] a, int i){ 
        left=2*i;
        right=2*i+1;
        if(left <= n && a[left] > a[i]){
            largest=left;
        }
        else{
            largest=i;
        }
        
        if(right <= n && a[right] > a[largest]){
            largest=right;
        }
        if(largest!=i){
            exchange(i,largest);
            maxheap(a, largest);
        }
    }

        int t=a[i];
        a[i]=a[j];
