    int getMaxNum(int[] a) {
        int ans = a[0];
        a[0] = a[n];
        n--;
        int father = 0;
        int temp;
        int son = father * d + 1;
        while (son<=n) {
            int j = son + 1;
            while(j<=n && j <= father * d + d ) {
                if (a[j] > a[son]) {
                    son = j;
                }
                j++;
            }
            if (a[son] > a[father]) {
                temp = a[son];
                a[son] = a[father];
                a[father] = temp;
                father = son;
                son = father * d + 1;
            } else break;
        }
        return ans;
    }


 private void setSort(int[] input ){
        for (int sq = 1; sq < input.length; sq += sq){
            for (int j =0;j<input.length-sq;j += sq*2){
                setMergeSort(input,j,j+sq-1,Math.min(j+sq+sq,input.length-1));
            }
        }
    }
    private void setMergeSort(int[] input,int start,int mid , int end){
        int[] tem = new int[input.length];

       for ( int i = start; i <= end ; i ++){
           tem[i] = input[i];
       }
        int i= start,j=mid +1;
       for (int h = start;h<= end;h++){
           if (i > mid){
               input[h] = tem[j++];
           }else if(j > end){
               input[h] = tem[i++];
           } else if (tem[i] < tem[j]){
               input[h] = tem[i++];
           }else if (tem[i] > tem[j]){
               input[h] = tem[j++];
           }
       }
    }
