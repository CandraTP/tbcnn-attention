    int getMaxNum(int[] a) {
        int ans = a[0];
        a[0] = a[n];
        n--;
        int father = 0;
        int temp;
        int son = father * d + 1;
        while (son<=n) {
            int j = son + 1;
            while(j<=n && j <= father * d + d ) {
                if (a[j] > a[son]) {
                    son = j;
                }
                j++;
            }
            if (a[son] > a[father]) {
                temp = a[son];
                a[son] = a[father];
                a[father] = temp;
                father = son;
                son = father * d + 1;
            } else break;
        }
        return ans;

public static int[] sortiereIterativ(int[] a) {
        
        int i = 0;
        int position = 0;
        
        for (int sortiert = 0; sortiert < a.length; sortiert++) {
            
            
            i = sortiert;
            position = sortiert;
            
            
            while (i < a.length) {
                if (a[i] < a[position]) position = i;
                i++;
            }
            
            
            a[0] = a[sortiert];
            a[sortiert] = a[position];
            a[position] = a[0]; 
        }
        
        a[0] = -1;
        
        return a;
    }
