void sort(double arr[])
    {
        int n = arr.length;
         for (int i = 0; i < n-1; i++)
        {
            int min_idx = i;
            for (int j = i+1; j < n; j++)
                if (arr[j] < arr[min_idx])
                    min_idx = j;
 

            double temp = arr[min_idx];
            arr[min_idx] = arr[i];
            arr[i] = temp;
        }
    }


public static void sort(int[] data) {
        for (int k = data.length - 1; k > 0; k--) {
            int largestIdx = 0;
            for (int i = 0; i <= k; i++) {
                if (data[i] > data[largestIdx]) {
                    largestIdx = i;
                }
            }
            int tmp = data[k];
            data[k] = data[largestIdx];
            data[largestIdx] = tmp;
        }
    }

}
