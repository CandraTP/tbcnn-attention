public void bubbleSortWhile(int[] array) {

        System.out.println(Arrays.toString(array));
        int aux;
        int i = array.length;

        while (i > 0) {
            int j = 0;
            while (j < i - 1) {
                if (array[j] > array[j + 1]) {
                    aux = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = aux;
                }
                j++;
                System.out.println(Arrays.toString(array));
            }
            i--;
        }
    }


private void insertionSort(int[] a, int len) {
        
        for (int i = 1; i < len; i++) {
            int value = a[i];   
            int position = i;   
            
            
            while (position > 0 && a[position-1] > value) {
                a[position] = a[position-1];    
                position--;                     
            }
            
            
            a[position] = value;
        }
    }
