public static void insertionSort(int[] list, int n) {
        int i, j, key;
        
        
        for (i = 1; i < n; i++) {
            key = list[i]; 
            
            
            
            
            for (j = i - 1; j >= 0 && list[j] > key; j--) {
                list[j+1] = list[j];
            }
            
            list[j+1] = key;
        }
    }


public static int partition(int arr[], int low, int high)
 {
     int pivot = arr[high]; 
     int i = (low-1); 
     for (int j=low; j<high; j++)
     {
         
         
         if (arr[j] <= pivot)
         {
             i++;

             
             int temp = arr[i];
             arr[i] = arr[j];
             arr[j] = temp;
         }
     }

     
     int temp = arr[i+1];
     arr[i+1] = arr[high];
     arr[high] = temp;

     return i+1;
 }


 
 public static void sort(int arr[], int low, int high)
 {
     if (low < high)
     {
         
         int pi = partition(arr, low, high);

         
         
         sort(arr, low, pi-1);
         sort(arr, pi+1, high);
     }
 }
