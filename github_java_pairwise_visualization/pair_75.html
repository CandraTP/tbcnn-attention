public static void shellSortMarca(Automovil A[], int numAle){
        int salto, j;
        Automovil aux;
        boolean cambios;
        for(salto=numAle/2; salto!=0; salto/=2){
            cambios=true;
            while(cambios==true){ 
                cambios=false;
                for(j=salto; j< numAle; j++) 
                    if((A[j-salto].getMarca().compareTo(A[j].getMarca()))>0){ 
                        aux=A[j]; 
                        A[j]=A[j-salto];
                        A[j-salto]=aux;
                        cambios=true; 
                    }
            }
        }
    }


public static void shellSort(int t[]) {
        
        int n = t.length;
        int h = 1;
        while (h <= n / 3) {     
            h = 3 * h + 1;
        }
        while (h > 0) {
            for (int i = h; i < n; i++) {                   
                for (int j = i; j>= h ; j-=h) {             
                    if (t[j] < t[j-h]) {
                        int temp = t[j];
                        t[j] = t[j-h];
                        t[j-h] = temp;
                    }
                }
            }           
            h = h / 3;
        }
    }
