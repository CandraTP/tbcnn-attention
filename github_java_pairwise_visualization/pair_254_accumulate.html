public int[] sorter(int[] array) {
        if (array.length <= 1) {
            return array;
        }
        int[] left = new int[array.length / 2];
        int[] right = new int[array.length - left.length];
        System.arraycopy(array, 0, left, 0, left.length);
        System.arraycopy(array, left.length, right, 0, right.length);

        sorter(left);
        sorter(right);
        merge(left, right, array);
        return array;


    }

    private static void merge(int[] left, int[] right, int[] array) {
        int firstIndex = 0;
        int secondIndex = 0;
        int union = 0;

        while (firstIndex < left.length && secondIndex < right.length) {
            if (left[firstIndex] <= right[secondIndex]) {
                array[union] = left[firstIndex];
                firstIndex++;
            } else {
                array[union] = right[secondIndex];
                secondIndex++;
            }
            union++;
        }

        System.arraycopy(left, firstIndex, array, union, left.length - firstIndex);
        System.arraycopy(right, secondIndex, array, union, right.length - secondIndex);
    }


 private void setSort(int[] input ){
        for (int sq = 1; sq < input.length; sq += sq){
            for (int j =0;j<input.length-sq;j += sq*2){
                setMergeSort(input,j,j+sq-1,Math.min(j+sq+sq,input.length-1));
            }
        }
    }
    private void setMergeSort(int[] input,int start,int mid , int end){
        int[] tem = new int[input.length];

       for ( int i = start; i <= end ; i ++){
           tem[i] = input[i];
       }
        int i= start,j=mid +1;
       for (int h = start;h<= end;h++){
           if (i > mid){
               input[h] = tem[j++];
           }else if(j > end){
               input[h] = tem[i++];
           } else if (tem[i] < tem[j]){
               input[h] = tem[i++];
           }else if (tem[i] > tem[j]){
               input[h] = tem[j++];
           }
       }
    }
