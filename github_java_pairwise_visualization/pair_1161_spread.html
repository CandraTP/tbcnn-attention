public static void bubbleSort(int[] arr) {
        int length = arr.length;
        int k;

        for (int i = length; i >= 0; i--) {
            for (int j = 0; j < length-1; j++) {
                k = j + 1;

                if (arr[j] > arr[k])
                    swap(j, k, arr);
            }
            printArray(arr);
        }
    }

  

    private static void swap(int j, int k, int[] arr) {
        int temp = arr[j];
        arr[j] = arr[k];
        arr[k] = temp;

    }


   public static void insertionSort(int[] list, int n) {
        int i, j, key;
        
        
        for (i = 1; i < n; i++) {
            key = list[i]; 
            
            
            
            
            for (j = i - 1; j >= 0 && list[j] > key; j--) {
                list[j+1] = list[j];
            }
            
            list[j+1] = key;
        }
    }
