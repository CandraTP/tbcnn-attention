public static int getmin(ArrayList<Integer> a){
	int retVal = a.get(0);
	a.set(0, a.get(a.size()-1));
	a.remove(a.size()-1);
	if(!a.isEmpty()) siftd(a,0);
	
	return retVal;
}

public static boolean verify(ArrayList<Integer> a){
	
	for(int i = 0; i < a.size()-1; i++){
		if(a.get(i) > a.get(i+1)) return false;
	}
	return true;
	
}


public static void heapify(ArrayList<Integer> a){
	for(int i = a.size()/2; i >=0 ; i--){
		siftd(a,i);
	}
	
}

public static void siftd(ArrayList<Integer> a, int i){
	int left = i*2;
	int right = 2*i +1;
	int smallest = i;
	
	if(left < a.size() && a.get(left) < a.get(i) ){
		smallest = left;
	}
	if(right < a.size() && a.get(right) < a.get(smallest) ){
		smallest = right;
	}
	if(smallest != i){
		int temp = a.get(i);
		a.set(i,a.get(smallest));
		a.set(smallest, temp);
		siftd(a,smallest);
	}
}
	





    private static int[] sort(int[] data, int start){
        if(start < data.length-1){
            swap(data, start, findMinIndex(data, start));
            sort(data,start+1);
        }
        return data;
    }
    
    private static int findMinIndex(int[] data, int start){
        int minpos=start;
        for(int i=start+1; i<data.length; i++){
            if(data[i]<data[minpos]){
                minpos=i;
            }
        }
        return minpos;
    }
    
    private static void swap(int[] data, int idx1, int idx2){
        int temp1 = data[idx1];
        int temp2 = data[idx2];
        data[idx1] = temp2;
        data[idx2] = temp1;
        
    }
