public void shellSort(int[] array) {
        for(int gap = array.length / 2; gap > 0; gap /= 2) {
            for(int i = 0; i < gap; i++) {
                for(int j = i; j < array.length; j += gap) {
                    int min = j;
                    for(int k = j + gap; k < array.length; k += gap) {
                        if(array[k] < array[min]) {
                            min = k;
                        }
                    }
                    swap(array, j, min);
                }
            }
        }
    }

    public void swap(int[] array, int a, int b) {
        int temp = array[a];
        array[a] = array[b];
        array[b] = temp;
    }


public static int[] sortiereIterativ(int[] a) {
        
        int i = 0;
        int position = 0;
        
        for (int sortiert = 0; sortiert < a.length; sortiert++) {
            
            
            i = sortiert;
            position = sortiert;
            
            
            while (i < a.length) {
                if (a[i] < a[position]) position = i;
                i++;
            }
            
            
            a[0] = a[sortiert];
            a[sortiert] = a[position];
            a[position] = a[0]; 
        }
        
        a[0] = -1;
        
        return a;
    }
