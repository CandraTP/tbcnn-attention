public void shellSort_1() {
        
        long begin = System.nanoTime();
        int step = 0;
        
        while(step <= n) {
            step = 3 * step + 1;
        }

        if (step == 1 || step == 4) {
            step = 1;
        }
        else {
            for (int i = 0; i < 2; i++) {
                step = (step - 1) / 3;
            }
        }
        
        
        for (int shell = step; shell > 0; shell = (shell - 1) / 3) {
            
            for (int i = shell; i < n; i += 1) {

                int temp = array[i];
                int j;
                
                for (j = i; j >= shell && array[j - shell] > temp; j -= shell) {
                    array[j] = array[j - shell];
                }

                array[j] = temp;
            }
        }
        
        long end = System.nanoTime();
        time = end - begin;
        
    } 
    


public int[] sort(int num[]) {
        
        int h = 1;
        
        while (h < num.length / 3)
            h = h * 3 + 1;
        for (; h > 0; h /= 3) {
            
            for (int k = 0; k < h; k++) {
                
                for (int i = h + k; i < num.length; i += h) {
                    int j = i;
                    
                    while (j >= h && num[j] < num[j - h]) {
                        swap(num, j, j - h);
                        j -= h;
                    }

                }
            }
        }

        return num;
    }

    
    private void swap(int arg[], int i, int j) {
        
        if (i != j) {
            arg[i] = arg[i] ^ arg[j];
            arg[j] = arg[j] ^ arg[i];
            arg[i] = arg[j] ^ arg[i];
        }
    }
