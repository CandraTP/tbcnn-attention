public static int partition(int arr[], int low, int high)
 {
     int pivot = arr[high]; 
     int i = (low-1); 
     for (int j=low; j<high; j++)
     {
         
         
         if (arr[j] <= pivot)
         {
             i++;

             
             int temp = arr[i];
             arr[i] = arr[j];
             arr[j] = temp;
         }
     }

     
     int temp = arr[i+1];
     arr[i+1] = arr[high];
     arr[high] = temp;

     return i+1;
 }


 
 public static void sort(int arr[], int low, int high)
 {
     if (low < high)
     {
         
         int pi = partition(arr, low, high);

         
         
         sort(arr, low, pi-1);
         sort(arr, pi+1, high);
     }
 }

 


 private void setSort(int[] input ){
        for (int sq = 1; sq < input.length; sq += sq){
            for (int j =0;j<input.length-sq;j += sq*2){
                setMergeSort(input,j,j+sq-1,Math.min(j+sq+sq,input.length-1));
            }
        }
    }
    private void setMergeSort(int[] input,int start,int mid , int end){
        int[] tem = new int[input.length];

       for ( int i = start; i <= end ; i ++){
           tem[i] = input[i];
       }
        int i= start,j=mid +1;
       for (int h = start;h<= end;h++){
           if (i > mid){
               input[h] = tem[j++];
           }else if(j > end){
               input[h] = tem[i++];
           } else if (tem[i] < tem[j]){
               input[h] = tem[i++];
           }else if (tem[i] > tem[j]){
               input[h] = tem[j++];
           }
       }
    }
