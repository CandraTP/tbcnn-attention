package org.burstsort4j;


public class Shellsort {

    private Shellsort() {
    }

    
    public static <T extends Comparable<? super T>> void sort(T[] input) {
        if (input == null || input.length < 2) {
            return;
        }

        int inc = input.length / 2;
        while (inc > 0) {
            for (int ii = inc; ii < input.length; ii++) {
                T temp = input[ii];
                int jj = ii;
                while (jj >= inc && input[jj - inc].compareTo(temp) > 0) {
                    input[jj] = input[jj - inc];
                    jj -= inc;
                }
                input[jj] = temp;
            }
            
            inc = inc == 2 ? 1 : inc * 5 / 11;
        }
    }
}



   public static void mergeSort(int[] arr) {
      if (arr.length > 1) {
         int[] left = leftHalf(arr);
         int[] right = rightHalf(arr);

         mergeSort(left);
         mergeSort(right);

         mergeArrays(arr, left, right);
      }
   }

   public static int[] leftHalf(int[] arr) {
      int size = arr.length / 2;
      int[] left = new int[size];
      for (int i = 0; i < size; i++) {
         left[i] = arr[i];
      }
      return left;
   }

   public static int[] rightHalf(int[] arr) {
      int size1 = arr.length / 2;
      int size2 = arr.length - size1;
      int[] right = new int[size2];
      for (int i = 0; i < size2; i++) {
         right[i] = arr[i + size1];
      }
      return right;
   }

   public static void mergeArrays(int[] arr, int[] left, int[] right) {
      int i1 = 0;
      int i2 = 0;

      for (int i = 0; i < arr.length; i++) {
         if (i2 >= right.length || (i1 < left.length && left[i1] <= right[i2])) {
            arr[i] = left[i1];
            i1++;
         } else {
            arr[i] = right[i2];
            i2++;
         }
      }
   }
