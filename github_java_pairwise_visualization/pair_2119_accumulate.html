public void shellSort(int[] array) {
        for(int gap = array.length / 2; gap > 0; gap /= 2) {
            for(int i = 0; i < gap; i++) {
                for(int j = i; j < array.length; j += gap) {
                    int min = j;
                    for(int k = j + gap; k < array.length; k += gap) {
                        if(array[k] < array[min]) {
                            min = k;
                        }
                    }
                    swap(array, j, min);
                }
            }
        }
    }

    public void swap(int[] array, int a, int b) {
        int temp = array[a];
        array[a] = array[b];
        array[b] = temp;
    }


public int[] sort(int num[]) {
        
        int h = 1;
        
        while (h < num.length / 3)
            h = h * 3 + 1;
        for (; h > 0; h /= 3) {
            
            for (int k = 0; k < h; k++) {
                
                for (int i = h + k; i < num.length; i += h) {
                    int j = i;
                    
                    while (j >= h && num[j] < num[j - h]) {
                        swap(num, j, j - h);
                        j -= h;
                    }

                }
            }
        }

        return num;
    }

    
    private void swap(int arg[], int i, int j) {
        
        if (i != j) {
            arg[i] = arg[i] ^ arg[j];
            arg[j] = arg[j] ^ arg[i];
            arg[i] = arg[j] ^ arg[i];
        }
    }
