public static int partition(int arr[], int low, int high)
 {
     int pivot = arr[high]; 
     int i = (low-1); 
     for (int j=low; j<high; j++)
     {
         
         
         if (arr[j] <= pivot)
         {
             i++;

             
             int temp = arr[i];
             arr[i] = arr[j];
             arr[j] = temp;
         }
     }

     
     int temp = arr[i+1];
     arr[i+1] = arr[high];
     arr[high] = temp;

     return i+1;
 }


 
 public static void sort(int arr[], int low, int high)
 {
     if (low < high)
     {
         
         int pi = partition(arr, low, high);

         
         
         sort(arr, low, pi-1);
         sort(arr, pi+1, high);
     }
 }

 


public static void selection_sort(int[]arr,int bi,int ei){
        if (bi==arr.length-1){
            return;
        }
        if (ei==arr.length){
            selection_sort(arr,bi+1,bi+2);
            return;
        }
        if (arr[bi]>arr[ei]){
            int temp = arr[bi];
            arr[bi] = arr[ei];
            arr[ei] = temp;
        
        }
        selection_sort(arr,bi,ei+1);
    }
