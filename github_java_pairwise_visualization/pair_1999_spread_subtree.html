public static void sort(Comparable[] a) {
        
        final int N  = a.length; int h = 1;
        while(h < N/3) h = 3*h + 1;
        
        
        while(h >= 1) {
        
            
            
            for(int i = h; i < N; i++) {
            
                
                
                for(int j = i; j >= h && less(a[j], a[j-h]); j -= h) exch(a, j, j-h);
                
            }
            h /= 3;
        }
        
        
    }


   public static void insertionSort(int[] list, int n) {
        int i, j, key;
        
        
        for (i = 1; i < n; i++) {
            key = list[i]; 
            
            
            
            
            for (j = i - 1; j >= 0 && list[j] > key; j--) {
                list[j+1] = list[j];
            }
            
            list[j+1] = key;
        }
    }
