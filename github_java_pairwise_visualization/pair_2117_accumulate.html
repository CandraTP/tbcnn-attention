public void bubbleSortWhile(int[] array) {

        System.out.println(Arrays.toString(array));
        int aux;
        int i = array.length;

        while (i > 0) {
            int j = 0;
            while (j < i - 1) {
                if (array[j] > array[j + 1]) {
                    aux = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = aux;
                }
                j++;
                System.out.println(Arrays.toString(array));
            }
            i--;
        }
    }


public static void selsort(int a[]){
        int n=a.length;
        int temp;
        int pos=0;
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                if(a[i]>a[j])
                    pos=j;
            }
           temp=a[i];
           a[i]=a[pos];
           a[pos]=temp;
        }
      
         System.out.println("the sorted array is ");
         for(int i=0;i<n;i++){
             System.out.print(a[i]+ " ");
         }
    }
