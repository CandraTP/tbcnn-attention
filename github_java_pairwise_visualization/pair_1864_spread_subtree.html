public static int[] sortiereIterativ(int[] a) {
        
        int i = 0;
        int position = 0;
        
        for (int sortiert = 0; sortiert < a.length; sortiert++) {
            
            
            i = sortiert;
            position = sortiert;
            
            
            while (i < a.length) {
                if (a[i] < a[position]) position = i;
                i++;
            }
            
            
            a[0] = a[sortiert];
            a[sortiert] = a[position];
            a[position] = a[0]; 
        }
        
        a[0] = -1;
        
        return a;
    }


    public static void selectionSortLargest(double[] array)
    {
        for (int r = array.length - 1; r > 0; --r)
        {
            int largestElementIndex = r;
            
            for (int l = 0; l < r; ++l)
            {
                if (array[l] > array[largestElementIndex])
                    largestElementIndex = l;
            }
            
            
            double temp = array[r];
            array[r] = array[largestElementIndex];
            array[largestElementIndex] = temp;
        }
    }
