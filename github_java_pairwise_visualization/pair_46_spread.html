package org.burstsort4j;


public class Shellsort {

    private Shellsort() {
    }

    
    public static <T extends Comparable<? super T>> void sort(T[] input) {
        if (input == null || input.length < 2) {
            return;
        }

        int inc = input.length / 2;
        while (inc > 0) {
            for (int ii = inc; ii < input.length; ii++) {
                T temp = input[ii];
                int jj = ii;
                while (jj >= inc && input[jj - inc].compareTo(temp) > 0) {
                    input[jj] = input[jj - inc];
                    jj -= inc;
                }
                input[jj] = temp;
            }
            
            inc = inc == 2 ? 1 : inc * 5 / 11;
        }
    }
}



public int[] sort(int num[]) {
        
        int h = 1;
        
        while (h < num.length / 3)
            h = h * 3 + 1;
        for (; h > 0; h /= 3) {
            
            for (int k = 0; k < h; k++) {
                
                for (int i = h + k; i < num.length; i += h) {
                    int j = i;
                    
                    while (j >= h && num[j] < num[j - h]) {
                        swap(num, j, j - h);
                        j -= h;
                    }

                }
            }
        }

        return num;
    }

    
    private void swap(int arg[], int i, int j) {
        
        if (i != j) {
            arg[i] = arg[i] ^ arg[j];
            arg[j] = arg[j] ^ arg[i];
            arg[i] = arg[j] ^ arg[i];
        }
    }
