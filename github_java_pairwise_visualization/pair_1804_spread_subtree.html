public void shellSort_1() {
        
        long begin = System.nanoTime();
        int step = 0;
        
        while(step <= n) {
            step = 3 * step + 1;
        }

        if (step == 1 || step == 4) {
            step = 1;
        }
        else {
            for (int i = 0; i < 2; i++) {
                step = (step - 1) / 3;
            }
        }
        
        
        for (int shell = step; shell > 0; shell = (shell - 1) / 3) {
            
            for (int i = shell; i < n; i += 1) {

                int temp = array[i];
                int j;
                
                for (j = i; j >= shell && array[j - shell] > temp; j -= shell) {
                    array[j] = array[j - shell];
                }

                array[j] = temp;
            }
        }
        
        long end = System.nanoTime();
        time = end - begin;
        
    } 
    


 
  public static void sort(double[] a)
  {
    int n = a.length;
    temp = new double[n];
    recursiveSort(a, 0, n-1);
  }

  
  private static void recursiveSort(double[] a, int from, int to)
  {
    if (to - from < 2)       
    {
      if (to > from && a[to] < a[from])
      {
        double aTemp = a[to];  
        a[to] = a[from];
        a[from] = aTemp;
      }
    }
    else                     
    {
      int middle = (from + to) / 2;
      recursiveSort(a, from, middle);
      recursiveSort(a, middle + 1, to);
      merge(a, from, middle, to);
    }
  }

  
  
  private static void merge(double[] a, int from, int middle, int to)
  {
    int i = from, j = middle + 1, k = from;

    
    while (i <= middle && j <= to)
    {
      if (a[i] < a[j])
      {
        temp[k] = a[i];   
        i++;
      }
      else
      {
        temp[k] = a[j];
        j++;
      }
      k++;
    }

    
    while (i <= middle)
    {
      temp[k] = a[i];     
      i++;
      k++;
    }

    
    while (j <= to)
    {
      temp[k] = a[j];     
      j++;
      k++;
    }

    
    for (k = from; k <= to; k++)
      a[k] = temp[k];
  }
