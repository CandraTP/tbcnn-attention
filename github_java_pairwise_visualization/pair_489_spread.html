public List<Integer> sort(List<Integer> list){

        
        
        
        int n = list.size();
        
        while(n != 0){
            
            int swappy = 0;
            for (int i = 1;i<n;i++){
                
                if(list.get(i-1) > list.get(i)){
                    int temp = list.get(i-1);
                    list.set(i-1,list.get(i));
                    list.set(i,temp);
                    
                    swappy = i;
                }
            }
            
            n = swappy;
        }
        return list;
    }


public static int partition(int arr[], int low, int high)
 {
     int pivot = arr[high]; 
     int i = (low-1); 
     for (int j=low; j<high; j++)
     {
         
         
         if (arr[j] <= pivot)
         {
             i++;

             
             int temp = arr[i];
             arr[i] = arr[j];
             arr[j] = temp;
         }
     }

     
     int temp = arr[i+1];
     arr[i+1] = arr[high];
     arr[high] = temp;

     return i+1;
 }


 
 public static void sort(int arr[], int low, int high)
 {
     if (low < high)
     {
         
         int pi = partition(arr, low, high);

         
         
         sort(arr, low, pi-1);
         sort(arr, pi+1, high);
     }
 }
