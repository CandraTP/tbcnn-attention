public static void quicksort(int[] a, int start, int end) {
        if (start >= end)
            return;

        int pavot = a[start];
        int i = start;
        int j = end;

        while (i < j) {
            while (i < j && a[j] > pavot)
                j--;
            while (i < j && a[i] <= pavot)
                i++;
            if (i != j) {
                int t = a[i];
                a[i] = a[j];
                a[j] = t;
            }
        }
        if (i == j) {
            int t = a[start];
            a[start] = a[i];
            a[j] = t;
        }
        quicksort(a, start, i - 1);
        quicksort(a, i + 1, end);
    }


 void sort(double arr[])
    {
        int n = arr.length;
         for (int i = 0; i < n-1; i++)
        {
            int min_idx = i;
            for (int j = i+1; j < n; j++)
                if (arr[j] < arr[min_idx])
                    min_idx = j;
 

            double temp = arr[min_idx];
            arr[min_idx] = arr[i];
            arr[i] = temp;
        }
    }
