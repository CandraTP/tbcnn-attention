private void insertionSort(int[] a, int len) {
        
        for (int i = 1; i < len; i++) {
            int value = a[i];   
            int position = i;   
            
            
            while (position > 0 && a[position-1] > value) {
                a[position] = a[position-1];    
                position--;                     
            }
            
            
            a[position] = value;
        }
    }
    


public static void selsort(int a[]){
        int n=a.length;
        int temp;
        int pos=0;
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                if(a[i]>a[j])
                    pos=j;
            }
           temp=a[i];
           a[i]=a[pos];
           a[pos]=temp;
        }
      
         System.out.println("the sorted array is ");
         for(int i=0;i<n;i++){

         }
    }
