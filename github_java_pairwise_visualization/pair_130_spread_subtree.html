int sort(int arr[])
    {
        int n = arr.length;
 
        
        for (int gap = n/2; gap > 0; gap /= 2)
        {
            
            
            
            
            for (int i = gap; i < n; i += 1)
            {
                
                
                
                int temp = arr[i];
 
                
                
                int j;
                for (j = i; j >= gap && arr[j - gap] > temp; j -= gap)
                    arr[j] = arr[j - gap];
 
                
                
                arr[j] = temp;
            }
        }
        return 0;
    }


   public static void insertionSort(int[] list, int n) {
        int i, j, key;
        
        
        for (i = 1; i < n; i++) {
            key = list[i]; 
            
            
            
            
            for (j = i - 1; j >= 0 && list[j] > key; j--) {
                list[j+1] = list[j];
            }
            
            list[j+1] = key;
        }
    }
