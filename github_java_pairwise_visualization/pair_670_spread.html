public static int partition(int arr[], int low, int high)
 {
     int pivot = arr[high]; 
     int i = (low-1); 
     for (int j=low; j<high; j++)
     {
         
         
         if (arr[j] <= pivot)
         {
             i++;

             
             int temp = arr[i];
             arr[i] = arr[j];
             arr[j] = temp;
         }
     }

     
     int temp = arr[i+1];
     arr[i+1] = arr[high];
     arr[high] = temp;

     return i+1;
 }


 
 public static void sort(int arr[], int low, int high)
 {
     if (low < high)
     {
         
         int pi = partition(arr, low, high);

         
         
         sort(arr, low, pi-1);
         sort(arr, pi+1, high);
     }
 }

 


   public static void mergeSort(int[] arr) {
      if (arr.length > 1) {
         int[] left = leftHalf(arr);
         int[] right = rightHalf(arr);

         mergeSort(left);
         mergeSort(right);

         mergeArrays(arr, left, right);
      }
   }

   public static int[] leftHalf(int[] arr) {
      int size = arr.length / 2;
      int[] left = new int[size];
      for (int i = 0; i < size; i++) {
         left[i] = arr[i];
      }
      return left;
   }

   public static int[] rightHalf(int[] arr) {
      int size1 = arr.length / 2;
      int size2 = arr.length - size1;
      int[] right = new int[size2];
      for (int i = 0; i < size2; i++) {
         right[i] = arr[i + size1];
      }
      return right;
   }

   public static void mergeArrays(int[] arr, int[] left, int[] right) {
      int i1 = 0;
      int i2 = 0;

      for (int i = 0; i < arr.length; i++) {
         if (i2 >= right.length || (i1 < left.length && left[i1] <= right[i2])) {
            arr[i] = left[i1];
            i1++;
         } else {
            arr[i] = right[i2];
            i2++;
         }
      }
   }
