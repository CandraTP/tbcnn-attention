void sort(double arr[])
    {
        int n = arr.length;
         for (int i = 0; i < n-1; i++)
        {
            int min_idx = i;
            for (int j = i+1; j < n; j++)
                if (arr[j] < arr[min_idx])
                    min_idx = j;
 

            double temp = arr[min_idx];
            arr[min_idx] = arr[i];
            arr[i] = temp;
        }
    }


import java.util.Arrays;
class SortingHeap{

void heapify(int arr[],int size,int i)
{
	int swapwith=i;
	int l=2*i+1;
	int r=2*i+2;
	
	
	
	if(l<size && arr[l]>arr[swapwith]) 
	{
		swapwith=l;
	}
	if(r<size && arr[r]>arr[swapwith]) 
	{
		swapwith=r;
	}	
	
	if(swapwith != i)
	{
		int tmp =arr[i];
		arr[i] = arr[swapwith];
		arr[swapwith] = tmp;

		heapify(arr,size,swapwith); 
	}	
		
}
