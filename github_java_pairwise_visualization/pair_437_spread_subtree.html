    private static void sort(int[] a, int lo, int hi) {
        
        if(hi <= lo)    return;
        int j = partition(a, lo, hi);
        sort(a, lo, j - 1);
        sort(a, j + 1, hi);
    }

    private static int partition(int[] a, int lo, int hi) {
        int i = lo, j = hi + 1;
        while(true){
            
            while(a[lo] > a[++i]) {
                if(i == hi) break;
            }
            
            while(a[lo] < a[--j]) {
                if(j == lo) break;
            }
            
            if(j <= i) break;
            
            exch(a, i, j);
        }
        
        exch(a, lo, j);
        return j;
    }

    private static void exch(int[] a, int i, int j) {
        int temp = a[i];
        a[i] = a[j];
        a[j] = temp;
    }

class SortingHeap{

void heapify(int arr[],int size,int i)
{
	int swapwith=i;
	int l=2*i+1;
	int r=2*i+2;
	
	
	
	if(l<size && arr[l]>arr[swapwith]) 
	{
		swapwith=l;
	}
	if(r<size && arr[r]>arr[swapwith]) 
	{
		swapwith=r;
	}	
	
	if(swapwith != i)
	{
		int tmp =arr[i];
		arr[i] = arr[swapwith];
		arr[swapwith] = tmp;

		heapify(arr,size,swapwith); 
	}	
		
}

{
	for(int i=(size/2)-1;i>=0;i--)

	}

		arr[i] = tmp;
