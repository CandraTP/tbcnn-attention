public static int[] doInsertionSort(int[] input){
         
        int temp;
        for (int i = 1; i < input.length; i++) {
            for(int j = i ; j > 0 ; j--){
                if(input[j] < input[j-1]){
                    temp = input[j];
                    input[j] = input[j-1];
                    input[j-1] = temp;
                }
            }
        }
        return input;
    }

        int pos=0;
        for(int i=0;i<n;i++){

                if(a[i]>a[j])
                    pos=j;
            }
           temp=a[i];
           a[i]=a[pos];

    }
