public static void sort(Comparable[] a) {
        
        final int N  = a.length; int h = 1;
        while(h < N/3) h = 3*h + 1;
        
        
        while(h >= 1) {
        
            
            
            for(int i = h; i < N; i++) {
            
                
                
                for(int j = i; j >= h && less(a[j], a[j-h]); j -= h) exch(a, j, j-h);
                
            }
            h /= 3;
        }
        
        
    }


public void bubbleSortWhile(int[] array) {

        System.out.println(Arrays.toString(array));
        int aux;
        int i = array.length;

        while (i > 0) {
            int j = 0;
            while (j < i - 1) {
                if (array[j] > array[j + 1]) {
                    aux = array[j];
                    array[j] = array[j + 1];
                    array[j + 1] = aux;
                }
                j++;
                System.out.println(Arrays.toString(array));
            }
            i--;
        }
    }
