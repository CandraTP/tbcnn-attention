private void setSort(int[] input ){
        for (int sq = 1; sq < input.length; sq += sq){
            for (int j =0;j<input.length-sq;j += sq*2){
                setMergeSort(input,j,j+sq-1,Math.min(j+sq+sq,input.length-1));
            }
        }
    }
    private void setMergeSort(int[] input,int start,int mid , int end){
        int[] tem = new int[input.length];

       for ( int i = start; i <= end ; i ++){
           tem[i] = input[i];
       }
        int i= start,j=mid +1;
       for (int h = start;h<= end;h++){
           if (i > mid){
               input[h] = tem[j++];
           }else if(j > end){
               input[h] = tem[i++];
           } else if (tem[i] < tem[j]){
               input[h] = tem[i++];
           }else if (tem[i] > tem[j]){
               input[h] = tem[j++];
           }
       }
    }

    public static void sort(int arr[])  
    {
            buildMaxHeap(arr);
            
            for(int i=arr.length-1; i>=0; i--){
                int temp=arr[0];
                arr[0] = arr[i];
                arr[i]= temp;

                n--;
                
                maxHeapify(arr,0);
		}
    }
    
    public static void maxHeapify(int[] arr, int i) 
    {
		int leftChild= (2*i)+1;
		int rightChild=(2*i)+2;
		
                int max;
                
                if((leftChild < n) && (arr[leftChild] > arr[i])) 
			max=leftChild;
                else max = i;
                
		if( (rightChild < n) && (arr[rightChild] > arr[max])) 
			max=rightChild;
		
                if(max!=i){
			int temp = arr[max];
			arr[max] = arr[i];
			arr[i] = temp;
                        maxHeapify(arr, max);
		}
	}  
    
    
    
    public static void buildMaxHeap(int [] array){
    
        n=array.length;
        
        for(int i=array.length/2; i>=0; i--){
            maxHeapify(array, i);
        }
    }
