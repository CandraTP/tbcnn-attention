public static void insertionSort(int[] list, int n) {
        int i, j, key;
        
        
        for (i = 1; i < n; i++) {
            key = list[i]; 
            
            
            
            
            for (j = i - 1; j >= 0 && list[j] > key; j--) {
                list[j+1] = list[j];
            }
            
            list[j+1] = key;
        }
    }


package algorithms.sort;

            int key = arr[i];

                arr[j+1] = arr[j];
                
                j--;
            }
            
            arr[j+1] = key;            
        }
