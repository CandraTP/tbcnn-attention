package org.hibernate.envers.internal.tools.graph;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.List;


public class TopologicalSort<R> {
	private List<R> sorted;
	private int time;

	private void process(Vertex<R> v) {
		if ( v.getStartTime() != 0 ) {
			
			return;
		}

		v.setStartTime( time++ );

		for ( Vertex<R> n : v.getNeighbours() ) {
			process( n );
		}

		v.setEndTime( time++ );

		sorted.add( v.getRepresentation() );
	}

	public List<R> sort(Collection<Vertex<R>> vertices) {
		sorted = new ArrayList<>( vertices.size() );

		time = 1;

		for ( Vertex<R> v : vertices ) {
			if ( v.getEndTime() == 0 ) {
				process( v );
			}
		}

		Collections.reverse( sorted );

		return sorted;
	}
}



 public static void bubbleSort(int arr[]) {
      int len = arr.length;
      for (int i = len; i >= 0; i--) {
         for (int j = 0; j < len - 1; j++) {
            int k = j + 1;
            if (arr[j] > arr[k]) {
               swapNumbers(j, k, arr);
            }
         }
      }
   }

   private static void swapNumbers(int i, int j, int[] arr) {
      int temp;
      temp = arr[i];
      arr[i] = arr[j];
      arr[j] = temp;
   }
