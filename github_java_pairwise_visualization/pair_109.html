private static int[] sort(char[] a) {
        int n = a.length;
        int[] order = new int[n];
        int[] count = new int[R + 1]; 
        char[] sorted = new char[n];
        
        for (int i = 0; i < n; i++) {
            count[((int) a[i]) + 1]++;
        }
        
        for (int i = 1; i <= R; i++) {
            count[i] += count[i - 1];
        }
        
        
        for (int i = 0; i < n; i++) {
            order[count[(int) a[i]]] = i;
            sorted[count[(int) a[i]]++] = a[i];            
        }
        
        for (int i = 0; i < n; i++) {
            a[i] = sorted[i];
        }        
        return order;
    }
    
    private static void sort(char[] a, int[] b) {
        assert(a.length == b.length);
        int n = a.length;
        int[] order = sort(a);
        int[] tmp = new int[n];
        for (int i = 0; i < n; i++) {
            tmp[i] = b[order[i]];
        }
        for (int i = 0; i < n; i++) {
            b[i] = tmp[i];
        }        
    }


package linear;

import java.util.Random;


public class RadixSort {
    
    public static int[] sort(int[] input){
        
        for(int place=1; place <= 1000000000; place *= 10){
            
            input = countingSort(input, place);
        }

        return input;
    }

    private static int[] countingSort(int[] input, int place){
        int[] out = new int[input.length];

        int[] count = new int[10];

        for(int i=0; i < input.length; i++){
            int digit = getDigit(input[i], place);
            count[digit] += 1;
        }

        for(int i=1; i < count.length; i++){
            count[i] += count[i-1];
        }

        for(int i = input.length-1; i >= 0; i--){
            int digit = getDigit(input[i], place);

            out[count[digit]-1] = input[i];
            count[digit]--;
        }

        return out;

    }

    private static int getDigit(int value, int digitPlace){
        return ((value/digitPlace ) % 10);
    }
