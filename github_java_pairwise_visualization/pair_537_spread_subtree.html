public void bubbleSort(){

            for(int j=0;j<length-1-i;j++){
                compare(j,j+1);
            }
        }
    }
    public void compare(int x, int y){
        int z=0;
        if(a[x]>a[y]){
            z=a[x];
            a[x]=a[y];
            a[y]=z;
        }   
    }



    public static void sort(Comparable[] a) {
        
        final int N  = a.length; int h = 1;
        while(h < N/3) h = 3*h + 1;
        
        
        while(h >= 1) {
        
            
            
            for(int i = h; i < N; i++) {
            
                
                
                for(int j = i; j >= h && less(a[j], a[j-h]); j -= h) exch(a, j, j-h);
                
            }
            h /= 3;
        }
        
        
    }
