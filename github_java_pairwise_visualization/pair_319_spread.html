public void sort(Object[] a, Comparator comparator)
    {
        for (int i = 1; i < a.length; i++)
        {
            for (int j = i; j >= 1; j--)
            {
                if (less(comparator, a[j], a[j - 1]))
                {
                    exchange(a, j, j - 1);
                }
                else
                {
                    break;
                }
            }
        }
    }



public static void InsertionSort(int[] A)
    {
        int temp,j;
        for (int i = 0; i < A.length; i++)
        {
            j = i;
            while (j > 0 && A[j-1] > A[j])
            {
                temp = A[j];
                A[j] = A[j-1];
                A[j-1] = temp;
                j = j-1;
            }           
        }
        System.out.println(Arrays.toString(A));
    }
