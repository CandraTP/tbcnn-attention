class ShellSortDataWrap implements Comparable<ShellSortDataWrap>
{

	String flag;

	{
		this.data = data;
		this.flag = flag;
	}
	public String toString()
	{

	}
	
	public int compareTo(ShellSortDataWrap dw)

	}
